运算符重载

```c++
class Person{
    public:
     Person() {};
     Person(int a, int b) {
         this->m_a = a;
         this->m_b = b;
     }
     Person jiafa(const Person& p) { 
         Person temp;
         temp.m_a = p.m_a + this->m_a;
         temp.m_b = p.m_b + this->m_b;
         return temp;
     }
     Person jiafa(const Person& a, const Person& b) {
         Person temp;
         temp.m_a = a.m_a + b.m_a;
         temp.m_b = a.m_b + b.m_b;
         return temp;
     }
     Person jiafa(const Person& a, int num) { 
        Person temp;
        temp.m_a = a.m_a + num;
        temp.m_b = a.m_b + num;
        return temp;
     }

    public:
     int m_a;
     int m_b;
}
```
